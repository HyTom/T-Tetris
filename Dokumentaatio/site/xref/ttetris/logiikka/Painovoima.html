6   /**
7    * Luo ja hallinnoi tetriksessä käytettyä painovoimaa, joka määrää kuinka monta
8    * palaa tetris-pala tippuu per ruutu. Painovoima laskee ja nousee sen mukaan,
9    * mikä on pelissä saavutettu Level. Painovoimat on kovakoodattu pelin sääntöjen
10   * mukaisiksi.
11   */
12  public class Painovoima {

54      /**
55      *Palauttaa Tetrispelin leveliä vastaavan painovoiman.
56      */
57      public int annaPainovoima(int level) {
58          int palautettava = 0;
59          int avain = 0;
60          for (Integer key : this.painovoimat.keySet()) {
61              if (level >= key & avain <= key) {
62                  avain = key;
63                  palautettava = this.painovoimat.get(key);
64              }
65          }
66          return palautettava;
67      }
