1   package ttetris.ui;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import ttetris.logiikka.Kaivo;
7   import ttetris.tetriminot.Pala;
8   
9   public class Piirtaja extends JPanel {
10  
11      private Kaivo kaivo;
12      private int koko;
13  
14      public Piirtaja() {
15          super.setBackground(Color.WHITE);
16          this.koko = 20;
17      }
18  
19      public void setKaivo(Kaivo kaivo) {
20          this.kaivo = kaivo;
21      }
22  
23      protected void paintComponent(Graphics g) {
24          super.paintComponent(g);
25          piirraSeinat(g);
26          piirraKaivo(g);
27      }
28  
29      private void piirraSeinat(Graphics g) {
30          g.setColor(Color.DARK_GRAY);
31          //Katto
32          g.fillRect(0, 0, (kaivo.getLeveys() + 2) * koko, koko);
33          //vasen seinä
34          g.fillRect(0, 0, koko, (kaivo.getKorkeus() + 1) * koko);
35          //oikea seinä
36          g.fillRect((kaivo.getLeveys() + 1) * koko, 0, koko, (kaivo.getKorkeus() + 1) * koko);
37          //lattia
38          g.fillRect(0, kaivo.getKorkeus() * koko, (kaivo.getLeveys() + 2) * koko, koko);
39      }
40  
41      private void piirraKaivo(Graphics g) {
42          for (int y = 1; y < this.kaivo.getKorkeus(); y++) {
43              for (int x = 0; x < this.kaivo.getLeveys(); x++) {
44                  Pala pala = this.kaivo.getRuudukko()[x][y];
45                  if (pala != null) {
46                      g.setColor(pala.getVari());
47                      g.fillRect((x + 1) * koko,
48                              y * koko, koko, koko);
49                  }
50              }
51          }
52      }
53  
54  }
